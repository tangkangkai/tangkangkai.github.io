public String getPermutation(int n, int k) {
	int[] num = new int[n];
	for (int i = 0; i < n; i++) {
		num[i] = i + 1;
	}
	
	for(int i = 1; i < k; i++ ) {
		nextPermutation(num);
	}
		
	StringBuilder sb = new StringBuilder();
	for(int i : num) {
		sb.append(i);
	}
	return sb.toString();
}
